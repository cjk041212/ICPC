const int M = 63;
ll p[M];
bool zero;
void insert(ll x)
{
	for (int i = M; i >= 0; i--)
		if (x >> i == 1)
			if (p[i] == 0) { p[i] = x; return; }
			else x ^= p[i];
	zero = true;
}
ll qmax()
{
	ll ans = 0;
	for (int i = M; i >= 0; i--)ans = max(ans, ans ^ p[i]);
	return ans;
}
